API

GET /api/players_pts

Return most recent players points statistics

return all { "rank", "player", "alliance", "points" }

GET /api/players_flt

Return most recent players fleet statistics

return all { "rank", "player", "alliance", "points" }

GET /api/players_res

Return most recent players research statistics

return all { "rank", "player", "alliance", "points" }

GET /api/player_pts/:player

Return last 30 days player points statistics

return last 30 days { "date", "rank", "player", "alliance", "points" }

GET /api/player_flt/:player

Return last 30 days player fleet statistics

return last 30 days { "date", "rank", "player", "alliance", "points" }

GET /api/player_res/:player

Return last 30 days player research statistics

return last 30 days { "date", "rank", "player", "alliance", "points" }

GET /api/player_gal/:player

Return most recent player planets coordinates

return { "galaxy", "system", "position", "name", "player", "alliance" }

GET /api/galaxy?galaxy={1-9}&system={1-499}

Return most recent galaxy view

return all { "galaxy", "system", "position", "name", "moon", "player", "alliance" }

GET /api/top

Return last 30 days top player points statistics

return top last 30 days { "date", "points" }

GET /api/ally_pts/:alliance

Return most recent alliance players points statistics

return alliance { "date", "rank", "player", "alliance", "points" }
